  • Get Transactions of a Quote Trace in Group Center
    uses gw.api.database.Query
uses gw.api.database.Relop
uses java.util.Date
uses org.json.JSONObject
uses org.json.JSONException
uses gw.api.database.IQueryResult

var transactionID = "XXXXXXXXXXXXXXXXXXXX"
var publicMessageID = "pc:XXXXXXX"
var showTransactionJSON = true

transJson(showTransactionJSON)

function transJson(flag : Boolean) {
  if (flag) {
    getTransactionReportByGroupID(transactionID)
  } else {
    getMessageDetailsByPublicID(publicMessageID)
  }
}

function getTransactionReportByGroupID(groupID : String) {
  print("Transaction Report: " + groupID)

  var results = getMessagesByGroupID(groupID)

  printTransactionReportByMessageResult(results)
}

function getMessagesByGroupID(groupID : String) : IQueryResult<GroupCenterMessage_Ext, GroupCenterMessage_Ext> {
  return Query.make(GroupCenterMessage_Ext)
      .compare(GroupCenterMessage_Ext#RequestId, Relop.Equals, groupID)
      .select()
      .orderByDescending(\row -> row.CreateTime)
}

function printTransactionReportByMessageResult(results : IQueryResult<GroupCenterMessage_Ext, GroupCenterMessage_Ext>) {
  print("Number of messages found: " + results.Count)
  print("State     " + "Public ID               " + "Creation Date                      " + "Ending Date                        " + "Method                    " + "Risk ID                                             " + "Job Number       " +  "UWIssue  " +  "Inconsistency")
  results?.each(\result -> {
    print(result.Status + "          " + result.PublicID + "          " + getFormatDate(result.CreateTime) + "          " + getFormatDate(result.LastUpdate) + "          " + result.Method + calculateSpaces(result.Method, 19) + result.RiskId + calculateSpaces(result.Method, 36) + extractJobNumber(result) + "          " + extractUWIssuwMessage(result) + "          " + extractInconsistency(result))
  })
}

function calculateSpaces(element : String, maximumLength : int) : String {
  var numberSpaces = maximumLength - element.length + 10
  var spaces = ""
  var it = 0

  while (it < numberSpaces) {
    spaces = spaces + " "
    it++
  }

  return spaces
}

function extractJobNumber(result : GroupCenterMessage_Ext) : String {
  var requestJson = new JSONObject(transformBlobToString(result.Response))
  var camp = "null"

  try {
    camp = requestJson?.getJSONObject("data")?.getJSONObject("result")?.getJSONObject("policyData")?.get("jobNumber").toString()
  } catch(exception : JSONException) {
    print(exception.Message)
  }
  return camp
}

function extractUWIssuwMessage(result : GroupCenterMessage_Ext) : String {
  var requestJson = new JSONObject(transformBlobToString(result.Response))
  var camp = "null"

  try {
    camp = requestJson?.getJSONObject("data")?.getJSONObject("result")?.getJSONObject("quotingData")?.getJSONArray("uWIssues").getJSONObject(0).get("description").toString()
  } catch(exception : JSONException) {
    print(exception.Message)
  }

  return camp
}

function extractInconsistency(result : GroupCenterMessage_Ext) : String {
  var requestJson = new JSONObject(transformBlobToString(result.Response))
  var camp = "null"

  try {
    camp = requestJson?.getJSONObject("data")?.getJSONObject("result")?.getJSONObject("quotingData")?.getJSONArray("inconsistencyList").getJSONObject(0).get("description").toString()
  } catch(exception : JSONException) {
    print(exception.Message)
  }

  return camp
}

function getMessageDetailsByPublicID(publicId: String) {
  var result = Query.make(GroupCenterMessage_Ext).compare(GroupCenterMessage_Ext#PublicID, Relop.Equals, publicId).select()
  
  printMessageDetailsByQueryResult(result)
}

function printMessageDetailsByQueryResult(results : IQueryResult<GroupCenterMessage_Ext, GroupCenterMessage_Ext>) {
  results?.each(\result -> {
    print("Messages: " + result.Status + " " + result.PublicID + " " + getFormatDate(result.CreateTime) + " " + getFormatDate(result.LastUpdate) + "     " + result.Method)
    print("\n----------------------Request----------------------------")
    print(transformBlobToString(result.Request))
    print("\n----------------------Response----------------------------")
    print(transformBlobToString(result.Response))
  })
}

function getFormatDate(date : Date) : String {
  return date.format("yyyy-MM-dd HH:mm:ss")
}

function transformBlobToString(blob: Blob): String {
  var payloadString = new String(org.apache.commons.io.IOUtils.toString(blob.toInputStream()))
  
  return payloadString
}

  • Cancellation Process

    Quote

    uses gw.api.database.Query
uses gw.api.database.Relop

var _policyNumber = "XXXXXXXXXXXX"
var _policiesToCancel = getPoliciesToCancel(_policyNumber)
var _cancellationDate = "17/05/2022 00:01:00".toDate() // Format: DD/MM/YYYY

_policiesToCancel.each( \ policyPeriod -> {
  gw.transaction.Transaction.runWithNewBundle(\bundle -> {
    bundle.add(policyPeriod)

    var slice = policyPeriod.getSlice(_cancellationDate)
    var canCancel = slice.Policy.canStartCancellation(_cancellationDate)

    print('EditEffectiveDate: ' + _cancellationDate.format('dd/MM/yyyy HH:mm:ss'))

    if(canCancel == null) {
      print("")
      print("¿Can Cancel?: " + canCancel)

      var aCancellation = new Cancellation()

      aCancellation.Source = CancellationSource.TC_CARRIER
      aCancellation.Description = "Cancellation Reason"
      aCancellation.CancelReasonCode = ReasonCode.TC_ERRORPROCESSING_EXT
      aCancellation.startJob(slice.Policy, _cancellationDate, CalculationMethod.TC_PRORATA)
      aCancellation.UserLogged_Ext = policyPeriod.Job.UserLogged_Ext

      print(slice.PolicyNumber + " was cancelled - Job Number: " + aCancellation.JobNumber)
    }else {
      print(slice.PolicyNumber + " can not be cancelled.")
    }
  }, 'su')
})

function getPoliciesToCancel(aPolicyNumber : String) : ArrayList<PolicyPeriod> {
  var policiesToCancel : ArrayList<PolicyPeriod> = {}
  var policy = Policy.finder.findPolicyByPolicyNumber(aPolicyNumber)
  var policies = Query.make(entity.Policy)
      .compare(Policy#GroupID_Ext, Relop.Equals, policy?.GroupID_Ext)
      .select()
      .orderByDescending(\aPolicy -> aPolicy.CreateTime)

  policies.each(\ aPolicy  -> policiesToCancel.add(aPolicy.LatestBoundPeriod))

  return policiesToCancel
}
